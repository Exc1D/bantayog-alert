```typescript
import { BigQuery } from '@google-cloud/bigquery'
import { Logging } from '@google-cloud/logging'

const bq = new BigQuery()
const logging = new Logging()

export const auditExportBatch = onSchedule(
  { schedule: 'every 5 minutes', region: 'asia-southeast1', timeZone: 'UTC' },
  async () => {
    const table = bq.dataset('bantayog_audit').table('batch_events')

      logName: e.metadata.logName,
      resource: JSON.stringify(e.metadata.resource),
      payload: JSON.stringify(e.data),
      timestamp: e.metadata.timestamp,
    }))
    try {
      await table.insert(rows)
    } catch (err) {
      console.warn('[audit-export-batch] insert failed', err)
    }
  },
)
```

`audit-export-health-check.ts` — scheduled every 10 minutes, writes `system_health/latest`:

```typescript
import { onSchedule } from 'firebase-functions/v2/scheduler'
import { getFirestore, FieldValue } from 'firebase-admin/firestore'
import { getMessaging } from 'firebase-admin/messaging'
import { BigQuery } from '@google-cloud/bigquery'

const bq = new BigQuery()

export const auditExportHealthCheck = onSchedule(
  { schedule: 'every 10 minutes', region: 'asia-southeast1', timeZone: 'UTC' },
  async () => {
    const db = getFirestore()
    const now = Date.now()

    // Query last streaming write timestamp
    const [streamRows] = await bq
      .dataset('bantayog_audit')
      .table('streaming_events')
      .query('SELECT MAX(occurredAt) as lastAt FROM bantayog_audit.streaming_events')
    const lastStreamMs = streamRows[0]?.lastAt?.value ? Number(streamRows[0].lastAt.value) : 0
    const streamingGapSeconds = Math.floor((now - lastStreamMs) / 1000)

    // Query last batch write timestamp
    const [batchRows] = await bq
      .dataset('bantayog_audit')
      .table('batch_events')
      .query('SELECT MAX(timestamp) as lastAt FROM bantayog_audit.batch_events')
    const lastBatchMs = batchRows[0]?.lastAt?.value
      ? new Date(batchRows[0].lastAt.value).getTime()
      : 0
    const batchGapSeconds = Math.floor((now - lastBatchMs) / 1000)

    const healthy = streamingGapSeconds < 60 && batchGapSeconds < 900
    await db.doc('system_health/latest').set({
      streamingGapSeconds,
      batchGapSeconds,
      healthy,
      checkedAt: FieldValue.serverTimestamp(),
    })

    if (!healthy) {
      // Alert via FCM to superadmin topic (best-effort)
      try {
        await getMessaging().send({
          topic: 'superadmin-alerts',
          notification: {
            title: 'Audit pipeline health alert',
            body: `Streaming gap: ${streamingGapSeconds}s · Batch gap: ${batchGapSeconds}s`,
          },
        })
      } catch {
        /* non-critical */
      }
    }
  },
)
```

```typescript
// Province-level daily performance metrics
const startOfDayMs = new Date(`${date}T00:00:00.000Z`).getTime()
const endOfDayMs = startOfDayMs + 86400000
const resolvedSnap = await db
  .collection('report_ops')
```

```typescript
// TotpEnrollmentPage.tsx — bare-bones client-side TOTP enrollment
// Steps: getSession → generateSecret → show QR/key → verify code → enroll
import { useState } from 'react'
import { multiFactor, TotpMultiFactorGenerator, getAuth } from 'firebase/auth'
import type { TotpSecret } from 'firebase/auth'

export function TotpEnrollmentPage() {
  const auth = getAuth()
  const [totpSecret, setTotpSecret] = useState<TotpSecret | null>(null)
  const [verificationCode, setVerificationCode] = useState('')
  const [enrolled, setEnrolled] = useState(false)
  const [error, setError] = useState<string | null>(null)

  async function handleGenerate() {
    setError(null)
    const user = auth.currentUser
    if (!user) return
    const session = await multiFactor(user).getSession()
    const secret = await TotpMultiFactorGenerator.generateSecret(session)
    setTotpSecret(secret)
  }

  async function handleEnroll() {
    setError(null)
    const user = auth.currentUser
    if (!user || !totpSecret) return
    try {
      const assertion = TotpMultiFactorGenerator.assertionForEnrollment(
        totpSecret,
        verificationCode,
      )
      await multiFactor(user).enroll(assertion, 'Authenticator')
      setEnrolled(true)
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Enrollment failed')
    }
  }

  if (enrolled) return <p>TOTP enrolled successfully.</p>

  return (
    <div>
      <h1>Enroll TOTP Authenticator</h1>
      {!totpSecret ? (
        <button onClick={handleGenerate}>Generate Secret</button>
      ) : (
        <>
          <p>Secret key: <code>{totpSecret.secretKey}</code></p>
          <p>QR URI: <code>{totpSecret.generateQrCodeUrl('Bantayog Alert', auth.currentUser?.email ?? 'admin')}</code></p>
          <input
            placeholder="6-digit code"
            value={verificationCode}
            onChange={e => setVerificationCode(e.target.value)}
          />
          <button onClick={handleEnroll}>Verify and Enroll</button>
        </>
      )}
      {error && <p role="alert">{error}</p>}
    </div>
  )
}
```

```typescript
import { randomBytes } from 'node:crypto'
import * as bcrypt from 'bcrypt'
import { initializeApp, cert } from 'firebase-admin/app'
import { getFirestore } from 'firebase-admin/firestore'

const COST_FACTOR = 12
const CONFIG_DOC = 'system_config/break_glass_config'

async function main() {
  initializeApp({ credential: cert(process.env.GOOGLE_APPLICATION_CREDENTIALS ?? '') })
  const db = getFirestore()

  const existing = await db.doc(CONFIG_DOC).get()
  if (existing.exists) {
    console.log('Break-glass config already exists. No changes made.')
    return
  }

  const codeA = randomBytes(16).toString('hex')
  const codeB = randomBytes(16).toString('hex')
  const [hashA, hashB] = await Promise.all([
    bcrypt.hash(codeA, COST_FACTOR),
    bcrypt.hash(codeB, COST_FACTOR),
  ])

  await db.doc(CONFIG_DOC).set({ hashedCodes: [hashA, hashB] })

  console.log('Break-glass codes seeded. Store these securely:')
  console.log(`Controller A: ${codeA}`)
  console.log(`Controller B: ${codeB}`)
  console.log('These values will NOT be shown again.')
}

main().catch(console.error)
```

```typescript
// functions/src/callables/break-glass.ts
import { onCall, HttpsError } from 'firebase-functions/v2/https'
import { getFirestore } from 'firebase-admin/firestore'
import { getAuth } from 'firebase-admin/auth'
import * as bcrypt from 'bcrypt'
import { randomUUID } from 'node:crypto'
import { requireAuth, requireMfaAuth } from './https-error'
import { streamAuditEvent } from '../services/audit-stream'

const FOUR_HOURS_MS = 4 * 60 * 60 * 1000

// Core function — testable without onCall wrapper
export async function initiateBreakGlassCore(
  db: ReturnType<typeof getFirestore>,
  adminAuth: ReturnType<typeof getAuth>,
  input: { codeA: string; codeB: string; reason: string },
  actor: { uid: string },
): Promise<{ sessionId: string }> {
  const configDoc = await db.doc('system_config/break_glass_config').get()
  if (!configDoc.exists) {
    throw new HttpsError('not-found', 'break_glass_config_missing')
  }
  const { hashedCodes } = configDoc.data() as { hashedCodes: string[] }
  if (!Array.isArray(hashedCodes) || hashedCodes.length < 2) {
    throw new HttpsError('failed-precondition', 'break_glass_config_invalid')
  }

  // Order-independent: each code must match a different hash
  const [matchA0, matchA1, matchB0, matchB1] = await Promise.all([
    bcrypt.compare(input.codeA, hashedCodes[0]),
    bcrypt.compare(input.codeA, hashedCodes[1]),
    bcrypt.compare(input.codeB, hashedCodes[0]),
    bcrypt.compare(input.codeB, hashedCodes[1]),
  ])
  const codesValid = (matchA0 && matchB1) || (matchA1 && matchB0)
  if (!codesValid) {
    throw new HttpsError('unauthenticated', 'break_glass_codes_invalid')
  }

  const sessionId = randomUUID()
  const now = Date.now()
  const expiresAt = now + FOUR_HOURS_MS

  await adminAuth.setCustomUserClaims(actor.uid, {
    breakGlassSession: true,
    breakGlassSessionId: sessionId,
    breakGlassExpiresAt: expiresAt,
  })

  await db.doc(`breakglass_events/${sessionId}`).set({
    sessionId,
    actorUid: actor.uid,
    action: 'initiated',
    reason: input.reason,
    sessionStartedAt: now,
    expiresAt,
    schemaVersion: 1,
  })

  void streamAuditEvent({
    eventType: 'break_glass_initiated',
    actorUid: actor.uid,
    sessionId,
    occurredAt: now,
  })

  return { sessionId }
}

export const initiateBreakGlass = onCall(
  { region: 'asia-southeast1', enforceAppCheck: true },
  async (request) => {
    requireAuth(request, ['superadmin'])
    requireMfaAuth(request)
    const db = getFirestore()
    const adminAuth = getAuth()
    return initiateBreakGlassCore(db, adminAuth, request.data, { uid: request.auth!.uid })
  },
)

export async function deactivateBreakGlassCore(
  db: ReturnType<typeof getFirestore>,
  adminAuth: ReturnType<typeof getAuth>,
  actor: { uid: string; claims: Record<string, unknown> },
): Promise<void> {
  const sessionId = actor.claims.breakGlassSessionId as string | undefined
  if (!sessionId) {
    throw new HttpsError('failed-precondition', 'no_active_break_glass_session')
  }

  const userRecord = await adminAuth.getUser(actor.uid)
  const currentClaims = userRecord.customClaims ?? {}
  const {
    breakGlassSession: _a,
    breakGlassSessionId: _b,
    breakGlassExpiresAt: _c,
    ...remainingClaims
  } = currentClaims as Record<string, unknown>

  await adminAuth.setCustomUserClaims(actor.uid, remainingClaims)
  await db.doc(`breakglass_events/${sessionId}`).update({
    action: 'deactivated',
    deactivatedAt: Date.now(),
  })

  void streamAuditEvent({
    eventType: 'break_glass_deactivated',
    actorUid: actor.uid,
    sessionId,
    occurredAt: Date.now(),
  })
}

export const deactivateBreakGlass = onCall(
  { region: 'asia-southeast1', enforceAppCheck: true },
  async (request) => {
    requireAuth(request, ['superadmin'])
    const db = getFirestore()
    const adminAuth = getAuth()
    await deactivateBreakGlassCore(db, adminAuth, {
      uid: request.auth!.uid,
      claims: request.auth!.token as Record<string, unknown>,
    })
  },
)
```

### 7.A Task 2: `sweepExpiredBreakGlassSessions` scheduled trigger

**Primary files:**

- Modify: `functions/src/index.ts`

**Deliverables:**

```typescript
import { onSchedule } from 'firebase-functions/v2/scheduler'
import { getFirestore } from 'firebase-admin/firestore'
import { getAuth } from 'firebase-admin/auth'
import { streamAuditEvent } from '../services/audit-stream'

export const sweepExpiredBreakGlassSessions = onSchedule(
  { schedule: 'every 5 minutes', region: 'asia-southeast1', timeZone: 'UTC' },
  async () => {
    const db = getFirestore()
    const adminAuth = getAuth()
    const now = Date.now()

    const snap = await db
      .collection('breakglass_events')
      .where('action', '==', 'initiated')
      .where('expiresAt', '<', now)
      .get()

    for (const doc of snap.docs) {
      const { actorUid, sessionId } = doc.data() as { actorUid: string; sessionId: string }
      try {
        const userRecord = await adminAuth.getUser(actorUid)
        const currentClaims = userRecord.customClaims ?? {}

          breakGlassSession: _a,
          breakGlassSessionId: _b,
          breakGlassExpiresAt: _c,
          ...remaining

        await adminAuth.setCustomUserClaims(actorUid, remaining)
        await doc.ref.update({ action: 'auto_expired', expiredAt: now })

          eventType: 'break_glass_auto_expired',
          actorUid,
          sessionId,
          occurredAt: now,
```

```typescript
export async function declareEmergencyCore(
  db: ReturnType<typeof getFirestore>,
  input: z.infer<typeof declareEmergencyInputSchema>,

}
```
