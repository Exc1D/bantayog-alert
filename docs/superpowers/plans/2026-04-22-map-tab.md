# Map Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Map Tab as the default home screen of the Citizen PWA — full-bleed Leaflet map with public verified incident pins, personal report pins (from localForage + requestLookup), and a PeekSheet/DetailSheet reveal experience.

**Architecture:** `CitizenShell` is a thin layout wrapper (TopBar + BottomNav). `MapTab` is a self-contained vertical slice owning all local state (`useState`, no Zustand), its own Firestore hook (`usePublicIncidents`), its own local-storage hook (`useMyActiveReports`), and its own child components. No URL routing for the map tab state — `selectedPin` and `sheetPhase` are component-local.

**Critical data model note:** `reporterUid` does NOT exist in `reports/{id}` — it is only in `report_private/{id}`. Citizens cannot query `reports` by `reporterUid`. `useMyActiveReports` reads from `localforage` (written at submit time) and calls the `requestLookup` Firebase callable for live status, exactly like `LookupScreen`. The `report_lookup/{publicRef}` document is readable by any authenticated user and contains `reportId` for deduplication.

**Tech Stack:** React 19, Vite, Leaflet + react-leaflet, localforage, Firestore SDK v12, CSS custom properties (no Tailwind), Vitest + happy-dom + Testing Library.

```ts
import type { ReportStatus } from '@bantayog/shared-types'

export type ReportAction = 'edit' | 'cancel' | 'request_correction'

const EDITABLE = new Set<ReportStatus>(['new', 'awaiting_verify'])
const CORRECTION = new Set<ReportStatus>([
  'verified',
  'assigned',
  'acknowledged',
  'en_route',
  'on_scene',
  'resolved',
  'reopened',
])

export function actionsFor(status: ReportStatus): ReportAction[] {
  if (EDITABLE.has(status)) return ['edit', 'cancel']
  if (CORRECTION.has(status)) return ['request_correction']
  return []
}
```

```ts
// apps/citizen-pwa/src/services/localForageReports.ts
import localforage from 'localforage'
import type { ReportType, Severity } from '@bantayog/shared-types'

export interface StoredReport {
  publicRef: string
  secret: string
  reportType: ReportType
  severity: Severity
  lat: number
  lng: number
  submittedAt: number
  reportId?: string
}

const KEY = 'bantayog:reports:v1'

export async function loadReports(): Promise<StoredReport[]> {
  return (await localforage.getItem<StoredReport[]>(KEY)) ?? []
}

export async function saveReport(report: Omit<StoredReport, 'reportId'>): Promise<void> {
  const all = await loadReports()
  const idx = all.findIndex((r) => r.publicRef === report.publicRef)
  if (idx >= 0) {
    all[idx] = { ...all[idx]!, ...report }
  } else {
    all.push(report)
  }
  await localforage.setItem(KEY, all)
}

export async function updateReportId(publicRef: string, reportId: string): Promise<void> {
  const all = await loadReports()
  const idx = all.findIndex((r) => r.publicRef === publicRef)
  if (idx < 0) return
  all[idx] = { ...all[idx]!, reportId }
  await localforage.setItem(KEY, all)
}
```

```ts
// apps/citizen-pwa/src/hooks/usePublicIncidents.ts
import { useState, useEffect } from 'react'
import { collection, query, where, orderBy, limit, onSnapshot } from 'firebase/firestore'
import { db } from '../services/firebase.js'
import type { PublicIncident, Filters } from '../components/MapTab/types.js'

function windowMs(w: Filters['window']): number {
  if (w === '24h') return 24 * 60 * 60 * 1000
  if (w === '7d') return 7 * 24 * 60 * 60 * 1000
  return 30 * 24 * 60 * 60 * 1000
}

export function usePublicIncidents(filters: Filters): {
  incidents: PublicIncident[]
  loading: boolean
  error: unknown
} {
  const [incidents, setIncidents] = useState<PublicIncident[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<unknown>(null)

  useEffect(() => {
    const cutoff = Date.now() - windowMs(filters.window)
    const q = query(
      collection(db(), 'reports'),
      where('visibilityClass', '==', 'public_alertable'),
      where('submittedAt', '>=', cutoff),
      orderBy('submittedAt', 'desc'),
      limit(100),
    )
    const unsub = onSnapshot(
      q,
      (snap) => {
        const all = snap.docs.map((d) => ({
          id: d.id,
          ...(d.data() as Omit<PublicIncident, 'id'>),
        }))
        const filtered =
          filters.severity === 'all' ? all : all.filter((doc) => doc.severity === filters.severity)
        setIncidents(filtered)
        setLoading(false)
      },
      (err) => {
        setError(err)
        setLoading(false)
      },
    )
    return unsub
  }, [filters.severity, filters.window])

  return { incidents, loading, error }
}
```

```ts
// apps/citizen-pwa/src/hooks/useMyActiveReports.ts
import { useState, useEffect } from 'react'
import { getDoc, doc } from 'firebase/firestore'
import { httpsCallable } from 'firebase/functions'
import { db, fns } from '../services/firebase.js'
import { loadReports, updateReportId } from '../services/localForageReports.js'
import type { MyReport } from '../components/MapTab/types.js'

interface LookupResult {
  status: string
  lastStatusAt: number
  municipalityLabel: string
}

export function useMyActiveReports(): {
  reports: MyReport[]
  loading: boolean
} {
  const [reports, setReports] = useState<MyReport[]>([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    let cancelled = false

    async function fetchAll(): Promise<void> {
      const stored = await loadReports()
      if (stored.length === 0) {
        if (!cancelled) setLoading(false)
        return
      }

      const lookup = httpsCallable<{ publicRef: string; secret: string }, LookupResult>(
        fns(),
        'requestLookup',
      )

      const resolved = await Promise.all(
        stored.map(async (entry) => {
          let reportId = entry.reportId

          // Read report_lookup to get reportId if not yet cached
          if (!reportId) {
            const snap = await getDoc(doc(db(), 'report_lookup', entry.publicRef))
            if (snap.exists()) {
              const data = snap.data() as { reportId?: string }
              reportId = data.reportId
              if (reportId) await updateReportId(entry.publicRef, reportId)
            }
          }

          try {
            const res = await lookup({ publicRef: entry.publicRef, secret: entry.secret })
            const info = res.data
            return {
              publicRef: entry.publicRef,
              reportType: entry.reportType,
              severity: entry.severity,
              lat: entry.lat,
              lng: entry.lng,
              submittedAt: entry.submittedAt,
              id: reportId,
              status: info.status as MyReport['status'],
              municipalityLabel: info.municipalityLabel,
              lastStatusAt: info.lastStatusAt,
            } satisfies MyReport
          } catch {
            return {
              publicRef: entry.publicRef,
              reportType: entry.reportType,
              severity: entry.severity,
              lat: entry.lat,
              lng: entry.lng,
              submittedAt: entry.submittedAt,
              id: reportId,
              status: 'queued',
            } satisfies MyReport
          }
        }),
      )

      if (!cancelled) {
        setReports(resolved)
        setLoading(false)
      }
    }

    void fetchAll()
    return () => {
      cancelled = true
    }
  }, [])

  return { reports, loading }
}
```

```tsx
    if (disabled) return
    const opts = SEVERITY_OPTS.map((o) => o.value)
    const idx = opts.indexOf(filters.severity)
    const next = opts[(idx + 1) % opts.length]!
    onChange({ ...filters, severity: next })
  }

  function handleWindowClick() {
    if (disabled) return
    const opts = WINDOW_OPTS.map((o) => o.value)
    const idx = opts.indexOf(filters.window)
    const next = opts[(idx + 1) % opts.length]!
    onChange({ ...filters, window: next })

        disabled={disabled}
        onClick={handleSeverityClick}
      />
      <Pill
        label={windowLabel}
        active={filters.window !== '24h'}
        disabled={disabled}
        onClick={handleWindowClick}
      />
    </div>
```

```tsx
// apps/citizen-pwa/src/components/MapTab/PeekSheet.tsx
import { useRef } from 'react'

interface SelectedPin {
  id: string
  type: 'incident' | 'myReport'
  label: string
}

interface Props {
  sheetPhase: 'hidden' | 'peek' | 'expanded'
  pin: SelectedPin | null
  onExpand: () => void
  onDismiss: () => void
}

const SWIPE_THRESHOLD = 50

export function PeekSheet({ sheetPhase, pin, onExpand, onDismiss }: Props) {
  const touchStartY = useRef<number | null>(null)

  if (sheetPhase === 'hidden' || !pin) return null

  function handleTouchStart(e: React.TouchEvent) {
    touchStartY.current = e.touches[0]?.clientY ?? null
  }

  function handleTouchEnd(e: React.TouchEvent) {
    if (touchStartY.current === null) return
    const delta = (e.changedTouches[0]?.clientY ?? 0) - touchStartY.current
    touchStartY.current = null
    if (delta > SWIPE_THRESHOLD) onDismiss()
  }

  return (
    <div
      data-testid="peek-sheet"
      onTouchStart={handleTouchStart}
      onTouchEnd={handleTouchEnd}
      style={{
        position: 'fixed',
        bottom: 88,
        left: 0,
        right: 0,
        height: 80,
        background: 'rgba(255,255,255,0.95)',

        flexDirection: 'column',
```

```tsx
      mapRef.current?.remove()
      mapRef.current = null
    }
  }, [])

  // Offline detection
  useEffect(() => {
    function onOnline() {
      setIsOffline(false)
    }
    function onOffline() {
      setIsOffline(true)
    }
    window.addEventListener('online', onOnline)
    window.addEventListener('offline', onOffline)
    return () => {
      window.removeEventListener('online', onOnline)
      window.removeEventListener('offline', onOffline)
    }
  }, [])

  const suppressed = new Set(myReports.flatMap((r) => (r.id ? [r.id] : [])))
  const visibleIncidents = incidents.filter((i) => !suppressed.has(i.id))

  const handleIncidentTap = useCallback((incident: PublicIncident) => {
    const label = `${INCIDENT_LABELS[incident.reportType] ?? incident.reportType} · ${incident.severity.charAt(0).toUpperCase() + incident.severity.slice(1)} · ${incident.barangayId}, ${incident.municipalityLabel}`
    setSelectedPin({ id: incident.id, type: 'incident', label })
    setSheetPhase('peek')
  }, [])

  const handleMyReportTap = useCallback((report: MyReport) => {
    const label = `★ ${INCIDENT_LABELS[report.reportType] ?? report.reportType} · Your Report`
    setSelectedPin({ id: report.publicRef, type: 'myReport', label })
    setSheetPhase('peek')
  }, [])

  const selectedIncident =
    selectedPin?.type === 'incident'
      ? (incidents.find((i) => i.id === selectedPin.id) ?? null)
      : null

  const selectedMyReport =
    selectedPin?.type === 'myReport'
      ? (myReports.find((r) => r.publicRef === selectedPin.id) ?? null)
      : null

  const showEmpty =
    !incidentsLoading &&
    !myReportsLoading &&
    !incidentsError &&
    visibleIncidents.length === 0 &&
    myReports.length === 0

  return (
    <div style={{ position: 'relative', width: '100%', height: '100%' }}>
      {/* Map container */}
      <div ref={mapElRef} style={{ width: '100%', height: '100%' }} />

      {/* Layers (rendered after map is ready) */}
      {mapReady && (
        <>
          <IncidentLayer
            map={mapRef.current}
            incidents={visibleIncidents}
            suppressedIds={suppressed}
            onPinTap={handleIncidentTap}
          />
          <MyReportLayer map={mapRef.current} reports={myReports} onPinTap={handleMyReportTap} />
        </>
      )}

      {/* FilterBar */}
      <FilterBar filters={filters} onChange={setFilters} disabled={isOffline} />

      {/* Empty state */}
      {showEmpty && (
        <div
          role="status"
          style={{
            position: 'absolute',
            top: '50%',
            left: '50%',
            transform: 'translate(-50%, -50%)',
            background: 'var(--color-surface-container-low)',
            borderRadius: 12,
            padding: '20px 24px',
            maxWidth: 280,
            textAlign: 'center',
            boxShadow: '0 4px 24px rgba(0,30,64,0.12)',
            zIndex: 10,
          }}
        >
          <p
            style={{
              fontFamily: "'Inter', sans-serif",
              fontSize: '0.875rem',
              margin: '0 0 12px',
              color: 'var(--color-on-surface-variant)',
            }}
          >
            No reported incidents in this area in the last {filters.window}.
          </p>
          {filters.severity !== 'all' && (
            <button
              onClick={() => setFilters((f) => ({ ...f, severity: 'all' }))}
              style={{
                border: 'none',
                background: 'var(--color-primary)',
                color: '#fff',
                borderRadius: 8,
                padding: '8px 16px',
                cursor: 'pointer',
                fontFamily: "'Inter', sans-serif",
                fontSize: '0.875rem',
              }}
            >
              Clear filters
            </button>
          )}
        </div>
      )}

      {/* Offline banner */}
      {isOffline && (
        <div
          role="alert"
          style={{
            position: 'absolute',
            bottom: 0,
            left: 0,
            right: 0,
            background: 'rgba(0,30,64,0.9)',
            color: '#fff',
            fontFamily: "'Inter', sans-serif",
            fontSize: '0.8rem',
            padding: '8px 16px',
            textAlign: 'center',
            zIndex: 30,
          }}
        >
          📶 Offline — map data may be outdated
        </div>
      )}

      {/* PeekSheet */}
      <PeekSheet
        sheetPhase={sheetPhase}
        pin={selectedPin}
        onExpand={() => setSheetPhase('expanded')}
        onDismiss={() => {
          setSheetPhase('hidden')
          setSelectedPin(null)
        }}
      />

      {/* DetailSheet — public incident */}
      {sheetPhase === 'expanded' && selectedIncident && (
        <DetailSheet
          mode="public"
          incident={selectedIncident}
          sheetPhase={sheetPhase}
          onClose={() => {
            setSheetPhase('hidden')
            setSelectedPin(null)
          }}
          onCollapse={() => setSheetPhase('peek')}
        />
      )}

      {/* DetailSheet — own report */}
      {sheetPhase === 'expanded' && selectedMyReport && (
        <DetailSheet
          mode="myReport"
          report={selectedMyReport}
          sheetPhase={sheetPhase}
          onClose={() => {
            setSheetPhase('hidden')
            setSelectedPin(null)
          }}
          onCollapse={() => setSheetPhase('peek')}
        />
      )}
    </div>
  )
}
```
